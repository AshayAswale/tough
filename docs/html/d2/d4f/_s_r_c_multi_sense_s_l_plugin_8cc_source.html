278 void SRCMultiSenseSL::UpdateStates()
279 {
280  common::Time curTime = this->world->GetSimTime();
281 
282  // get imu data from imu link
283  if (this->imuSensor)
284  {
285  sensor_msgs::Imu imuMsg;
286  imuMsg.header.frame_id = this->imuLinkName;
287  imuMsg.header.stamp = ros::Time(curTime.Double());
288 
289  // compute angular rates
290  {
291 # if GAZEBO_MAJOR_VERSION >= 7
292  math::Vector3 wLocal = this->imuSensor->AngularVelocity();
293 # else
294  math::Vector3 wLocal = this->imuSensor->GetAngularVelocity();
295 # endif
296  imuMsg.angular_velocity.x = wLocal.x;
297  imuMsg.angular_velocity.y = wLocal.y;
298  imuMsg.angular_velocity.z = wLocal.z;
299  }
300 
301  // compute acceleration
302  {
303 # if GAZEBO_MAJOR_VERSION >= 7
304  math::Vector3 accel = this->imuSensor->LinearAcceleration();
305 # else
306  math::Vector3 accel = this->imuSensor->GetLinearAcceleration();
307 # endif
308  imuMsg.linear_acceleration.x = accel.x;
309  imuMsg.linear_acceleration.y = accel.y;
310  imuMsg.linear_acceleration.z = accel.z;
311  }
312 
313  // compute orientation
314  {
315  math::Quaternion imuRot =
316 # if GAZEBO_MAJOR_VERSION >= 7
317  this->imuSensor->Orientation();
318 # else
319  this->imuSensor->GetOrientation();
320 # endif
321  imuMsg.orientation.x = imuRot.x;
322  imuMsg.orientation.y = imuRot.y;
323  imuMsg.orientation.z = imuRot.z;
324  imuMsg.orientation.w = imuRot.w;
325  }
326 
327  this->pubImuQueue->push(imuMsg, this->pubImu);
328  }
329 
330  double dt = (curTime - this->lastTime).Double();
331  if (dt > 0)
332  {
333  this->jointStates.header.stamp = ros::Time(curTime.sec, curTime.nsec);
334  this->jointStates.name[0] = this->spindleJoint->GetName();
335  this->jointStates.position[0] = this->spindleJoint->GetAngle(0).Radian();
336  this->jointStates.velocity[0] = this->spindleJoint->GetVelocity(0);
337  this->jointStates.effort[0] = 0;
338 
339  if (this->spindleOn)
340  {
341  // PID control (velocity) spindle
342  double spindleError = this->spindleJoint->GetVelocity(0)
343  - this->spindleSpeed;
344  double spindleCmd = this->spindlePID.Update(spindleError, dt);
345  this->spindleJoint->SetForce(0, spindleCmd);
346 
347  this->jointStates.effort[0] = spindleCmd;
348 
349  this->lastTime = curTime;
350  }
351  else
352  {
353  this->spindlePID.Reset();
354  }
355  this->pubJointStatesQueue->push(this->jointStates, this->pubJointStates);
356  }
357 }

410 void SRCMultiSenseSL::SetMultiCameraFrameRate(const std_msgs::Float64::ConstPtr
411  &_msg)
412 {
413  // limit frame rate to what is capable
414  this->multiCameraFrameRate = static_cast<double>(_msg->data);
415 
416  // FIXME: Hardcoded lower limit on all resolution
417  if (this->multiCameraFrameRate < 1.0)
418  {
419  ROS_INFO("Camera rate cannot be below 1Hz at any resolution\n");
420  this->multiCameraFrameRate = 1.0;
421  }
422 
423  // FIXME: Hardcoded upper limit. Need to switch rates between modes.
424  if (this->imagerMode == 0)
425  {
426  if (this->multiCameraFrameRate > 15.0)
427  {
428  ROS_INFO("Camera rate cannot be above 15Hz at this resolution\n");
429  this->multiCameraFrameRate = 15.0;
430  }
431  }
432  else if (this->imagerMode == 1)
433  {
434  if (this->multiCameraFrameRate > 30.0)
435  {
436  ROS_INFO("Camera rate cannot be above 30Hz at this resolution\n");
437  this->multiCameraFrameRate = 30.0;
438  }
439  }
440  else if (this->imagerMode == 2)
441  {
442  if (this->multiCameraFrameRate > 60.0)
443  {
444  ROS_INFO("Camera rate cannot be above 60Hz at this resolution\n");
445  this->multiCameraFrameRate = 60.0;
446  }
447  }
448  else if (this->imagerMode == 3)
449  {
450  if (this->multiCameraFrameRate > 70.0)
451  {
452  ROS_INFO("Camera rate cannot be above 70Hz at this resolution\n");
453  this->multiCameraFrameRate = 70.0;
454  }
455  }
456  else
457  {
458  ROS_ERROR("MultiSense SL internal state error (%d)", this->imagerMode);
459  }
460 
461  this->multiCameraSensor->SetUpdateRate(this->multiCameraFrameRate);
462 }

465 void SRCMultiSenseSL::SetMultiCameraResolution(
466  const std_msgs::Int32::ConstPtr &_msg)
467 {
469  if (_msg->data < 0 || _msg->data > 3)
470  {
471  ROS_WARN("set_camera_resolution_mode must"
472  " be between 0 - 3:\n"
473  " 0 - 2MP (2048*1088) @ up to 15 fps\n"
474  " 1 - 1MP (2048*544) @ up to 30 fps\n"
475  " 2 - 0.5MP (1024*544) @ up to 60 fps (default)\n"
476  " 3 - VGA (640*480) @ up to 70 fps\n");
477  return;
478  }
479 
480  this->imagerMode = _msg->data;
481 
482  unsigned int width = 640;
483  unsigned int height = 480;
484  if (this->imagerMode == 0)
485  {
486  width = 2048;
487  height = 1088;
488  if (this->multiCameraFrameRate > 15)
489  {
490  ROS_INFO("Reducing frame rate to 15Hz.");
491  this->multiCameraFrameRate = 15.0;
492  }
493  }
494  else if (this->imagerMode == 1)
495  {
496  width = 2048;
497  height = 544;
498  if (this->multiCameraFrameRate > 30)
499  {
500  ROS_INFO("Reducing frame rate to 30Hz.");
501  this->multiCameraFrameRate = 30.0;
502  }
503  }
504  else if (this->imagerMode == 2)
505  {
506  width = 1024;
507  height = 544;
508  if (this->multiCameraFrameRate > 60)
509  {
510  ROS_INFO("Reducing frame rate to 60Hz.");
511  this->multiCameraFrameRate = 60.0;
512  }
513  }
514  else if (this->imagerMode == 3)
515  {
516  width = 640;
517  height = 480;
518  if (this->multiCameraFrameRate > 70)
519  {
520  ROS_INFO("Reducing frame rate to 70Hz.");
521  this->multiCameraFrameRate = 70.0;
522  }
523  }
524 
525  this->multiCameraSensor->SetUpdateRate(this->multiCameraFrameRate);
526 
527 # if GAZEBO_MAJOR_VERSION >= 7
528  for (unsigned int i = 0; i < this->multiCameraSensor->CameraCount(); ++i)
529  {
530  this->multiCameraSensor->Camera(i)->SetImageWidth(width);
531  this->multiCameraSensor->Camera(i)->SetImageHeight(height);
532  }
533 # else
534  for (unsigned int i = 0; i < this->multiCameraSensor->GetCameraCount(); ++i)
535  {
536  this->multiCameraSensor->GetCamera(i)->SetImageWidth(width);
537  this->multiCameraSensor->GetCamera(i)->SetImageHeight(height);
538  }
539 # endif
