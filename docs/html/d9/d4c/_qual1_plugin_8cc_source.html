36 void Qual1Plugin::Load(physics::WorldPtr _world, sdf::ElementPtr _sdf)
37 {
38  char *homePath = getenv("HOME");
39  if (!homePath)
40  {
41  gzerr << "Unable to get HOME environment variable. Report this error to "
42  << "an SRC official.\n";
43  return;
44  }
45  std::string logPath = homePath;
46  logPath += "/src_qual1_" + common::Time::GetWallTimeAsISOString() + ".log";
47 
48  this->logStream.open(logPath.c_str(), std::ios::out);
49  if (!this->logStream.is_open())
50  {
51  gzerr << "Unable to open log file[" << logPath << "].\n"
52  << "Make sure permissions are set correctly, and then retry.\n";
53  return;
54  }
55 
56  // Output header information
57  this->Log("# switch <light_index> <r> <g> <b> <a> <sim_sec> <sim_nsec>",
58  false);
59  this->Log("# answer <x> <y> <z> <r> <g> <b> <sim_sec> <sim_nsec>", false);
60 
61  this->world = _world;
62 
63  this->node = transport::NodePtr(new gazebo::transport::Node());
64  this->node->Init();
65 
66  // Wait for a subscriber to connect
67  this->pub = this->node->Advertise<gazebo::msgs::Visual>("~/visual");
68  this->pub->WaitForConnection();
69 
70  // Make sure the ROS node for Gazebo has already been initialized
71  if (!ros::isInitialized())
72  {
73  ROS_FATAL_STREAM("A ROS node for Gazebo has not been initialized,"
74  << "unable to load plugin. Load the Gazebo system plugin "
75  << "'libgazebo_ros_api_plugin.so' in the gazebo_ros package)");
76  return;
77  }
78 
79  // Load SDF parameters.
80  std::string robotNamespace = "";
81  if (_sdf->HasElement("robot_namespace"))
82  {
83  robotNamespace = _sdf->GetElement("robot_namespace")->Get<std::string>() +
84  "/";
85  }
86 
87  this->rosnode.reset(new ros::NodeHandle(robotNamespace));
88 
89  this->lightSub = this->rosnode->subscribe("/srcsim/qual1/start", 10,
90  &Qual1Plugin::OnStart, this);
91  this->startSub = this->rosnode->subscribe("/srcsim/qual1/light", 10,
92  &Qual1Plugin::OnLight, this);
93 
94  this->prevLightTime = _world->GetSimTime();
95 
96  int maxLight = 43;
97  int numLightSwitches = 10;
98  common::Time onDelay(1, 0);
99  common::Time offDelay(5, 0);
100 
101  if (_sdf->HasElement("switch_count"))
102  numLightSwitches = _sdf->Get<int>("switch_count");
103 
104  if (_sdf->HasElement("on_delay"))
105  onDelay.sec = _sdf->Get<int>("on_delay");
106 
107  this->lightPattern.push_back({1, 44, {5, 0}, gazebo::common::Color::White});
108 
109  std::vector<gazebo::common::Color> colors =
110  {
111  gazebo::common::Color::Red,
112  gazebo::common::Color::Blue,
113  gazebo::common::Color::Green,
114  };
115 
116  // Generate random lights
117  for (int i = 0; i < numLightSwitches; ++i)
118  {
119  int light = ignition::math::Rand::IntUniform(1, maxLight);
120  int console = 1;
121 
122  int colorIdx = ignition::math::Rand::IntUniform(0, colors.size() - 1);
123 
124  offDelay.sec = ignition::math::Rand::IntUniform(5, 10);
125  this->lightPattern.push_back({console, light, onDelay,
126  colors[colorIdx]});
127  this->lightPattern.push_back({console, light, offDelay,
128  gazebo::common::Color::Black});
129  }
130  this->lightPattern.push_back({1, 44, {5, 0}, gazebo::common::Color::Black});
131 
132  this->lightPatternIter = this->lightPattern.begin();
133 
134 }

137 void Qual1Plugin::Switch(int _light, const gazebo::common::Color &_clr)
138 {
139  std::ostringstream name, parent;
140  name << "console1::visuals::light" << _light;
141  parent << "console1::visuals";
142 
143  // Construct light visual message.
144  gazebo::msgs::Visual msg;
145  msg.set_name(name.str());
146  msg.set_parent_name(parent.str());
147  gazebo::msgs::Set(msg.mutable_material()->mutable_ambient(), _clr);
148  gazebo::msgs::Set(msg.mutable_material()->mutable_diffuse(), _clr);
149  gazebo::msgs::Set(msg.mutable_material()->mutable_emissive(), _clr);
150 
151  // Change the light
152  this->pub->Publish(msg);
153 }

190 void Qual1Plugin::OnUpdate()
191 {
192  // Stop when the light pattern reaches the end
193  if (this->lightPatternIter == this->lightPattern.end())
194  return;
195 
196  std::lock_guard<std::mutex> lock(this->iterMutex);
197 
198  // Compute the time difference.
199  gazebo::common::Time dt = this->world->GetSimTime() - this->prevLightTime;
200 
201  // Update the lights.
202  if (dt >= (*this->lightPatternIter).delay)
203  {
204  // Log light change data
205  std::ostringstream stream;
206  stream << "switch " << (*this->lightPatternIter).light << " "
207  << (*this->lightPatternIter).color;
208  this->Log(stream.str(), true);
209 
210  // Switch the light
211  this->Switch((*this->lightPatternIter).light,
212  (*this->lightPatternIter).color);
213 
214  this->lightPatternIter++;
215  this->prevLightTime = this->world->GetSimTime();
216  }
217 }
